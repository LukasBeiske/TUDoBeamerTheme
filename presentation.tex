% This adds a circle with a picture in it.
% Usage:

\newcommand{\roundpic}[6][]{
  \node [circle, draw, color=tugreen, minimum width = #2,
    path picture = {
      \node [#1] at (path picture bounding box.center) {
        \includegraphics[width=#3]{#4}
      };
    }] at (#5,#6) {};
}
